import express from "express"; import cors from "cors"; import fetch from "node-fetch"; const app = express(); app.use(cors()); app.use(express.json()); // ========================= // CONFIG // ========================= const ACCESS_TOKEN = process.env.WXCC_TOKEN; const ORG_ID = "c2e0792b-e4ea-4025-b456-7edc6d1c92cb"; const BASE_URL = "https://api.wxcc-eu2.cisco.com"; // ========================= // HEALTH CHECK // ========================= app.get("/health", (req, res) => { res.json({ status: "ok", service: "wxcc-backend" }); }); // ========================= // GET ENTRY POINT (FIXED) // ========================= app.get("/entrypoint/:id", async (req, res) => { const entryPointId = req.params.id; const url = `${BASE_URL}/organization/${ORG_ID}/entry-point/${entryPointId}`; try { const response = await fetch(url, { method: "GET", headers: { Authorization: `Bearer ${ACCESS_TOKEN}`, "Content-Type": "application/json" } }); const text = await response.text(); let data; try { data = JSON.parse(text); } catch { return res.status(500).json({ success: false, error: "Webex returned non-JSON response", raw: text }); } return res.status(response.status).json(data); } catch (err) { return res.status(500).json({ success: false, error: err.message }); } }); // ========================= // PUT ENTRY POINT UPDATE (SAFE) // ========================= app.put("/entrypoint/:id", async (req, res) => { const entryPointId = req.params.id; const { EmergencyCase, EmergencyPrompt } = req.body; const url = `${BASE_URL}/organization/${ORG_ID}/entry-point/${entryPointId}`; try { // 1. GET current entry point const getRes = await fetch(url, { headers: { Authorization: `Bearer ${ACCESS_TOKEN}` } }); const getText = await getRes.text(); let entryPoint; try { entryPoint = JSON.parse(getText); } catch { return res.status(500).json({ success: false, step: "GET_PARSE_ERROR", raw: getText }); } if (!getRes.ok) { return res.status(getRes.status).json({ success: false, step: "GET_FAILED", data: entryPoint }); } // 2. update flowOverrideSettings safely const updatedOverrides = (entryPoint.flowOverrideSettings || []).map(item => { if (item.name === "EmergencyCase") { return { ...item, value: String(!!EmergencyCase) }; } if (item.name === "EmergencyPrompt") { return { ...item, value: EmergencyPrompt || "" }; } return item; }); // 3. build full payload const updatedPayload = { ...entryPoint, flowOverrideSettings: updatedOverrides }; // remove read-only fields (IMPORTANT) delete updatedPayload.id; delete updatedPayload.createdTime; delete updatedPayload.lastUpdatedTime; // 4. PUT back to Webex const putRes = await fetch(url, { method: "PUT", headers: { Authorization: `Bearer ${ACCESS_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify(updatedPayload) }); const putText = await putRes.text(); let putData; try { putData = JSON.parse(putText); } catch { putData = { raw: putText }; } return res.status(putRes.status).json({ success: putRes.ok, step: "PUT", data: putData }); } catch (err) { return res.status(500).json({ success: false, error: err.message }); } }); // ========================= // START SERVER // ========================= const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`🚀 WXCC Backend running on port ${PORT}`); });
